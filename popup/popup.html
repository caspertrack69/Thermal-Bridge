    SDK ringan untuk integrasi web app POS ke printer thermal BLE. Gunakan connect() untuk pairing awal, lalu print() atau raw().

    Stable API Promise-based Event-ready
    Wajib dipanggil dari interaksi pengguna (klik tombol) saat pairing pertama.
    Method Reference 
    window.ThermalBridge.connect()    // Promise<{ success, device }>
window.ThermalBridge.disconnect() // Promise<{ success }>
window.ThermalBridge.print(data)  // Promise<{ success, message? }>
window.ThermalBridge.raw(bytes)   // Promise<{ success, message? }>
window.ThermalBridge.status()     // Promise<{ success, status, device? }>
    Quick Start (Recommended) 
    async function printReceipt(payload) {
  if (!window.ThermalBridge) {
    throw new Error('ThermalBridge extension tidak terdeteksi.');
  }

  const { status } = await window.ThermalBridge.status();
  if (status !== 'connected') {
    await window.ThermalBridge.connect(); // user click context
  }

  const result = await window.ThermalBridge.print(payload);
  if (!result.success) {
    throw new Error(result.message || 'Print failed');
  }
  return result;
}

window.addEventListener('thermalbridge:status', (event) => {
  const { status, device } = event.detail;
  console.log('[ThermalBridge]', status, device?.name || '-');
});
    Error Handling 
    try {
  await printReceipt(payload);
} catch (error) {
  const message = String(error?.message || error);
  if (/not detected/i.test(message)) {
    alert('Install/aktifkan extension Thermal Bridge.');
  } else if (/pair|connect|bluetooth/i.test(message)) {
    alert('Pastikan printer aktif, dekat, lalu ulangi pairing.');
  } else {
    alert('Cetak gagal: ' + message);
  }
}
